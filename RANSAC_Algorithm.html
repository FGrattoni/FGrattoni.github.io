outlier_mask = np.logical_not(inlier_mask)

plt.scatter(X[inlier_mask], y[inlier_mask],
	c = 'steelblue', edgecolor='white',
	marker='o', label='Inliers')
plt.scatter(X[outlier_mask], y[outlier_mask],
	c='red', edgecolor='white',
	marker='s', label='Outliers')

line_X = np.arange(min(X),max(X), 1)
line_y_ransac = ransac.predict(line_X[:, np.newaxis])
line_y_linear = linear.predict(line_X[:, np.newaxis])
plt.plot(line_X, line_y_ransac, color='green', lw=2)
plt.plot(line_X, line_y_linear, color='orange', lw=2)
plt.xlabel('', fontsize=15)
plt.ylabel('', fontsize=15)
plt.legend(labels=["RANSAC", "Simple Linear", "Inliers", "Outliers"], loc='upper right', fontsize=12)
