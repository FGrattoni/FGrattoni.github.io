#@title Seleziona il grado del polinomio: { run: "auto", vertical-output: true }
grado_polinomio = 3 #@param {type:"slider", min:1, max:20, step:1}


polyN = PolynomialFeatures(degree=grado_polinomio)
X_polyN_train = polyN.fit_transform(X_train.values.reshape(-1, 1))
polyN_reg = LinearRegression()
polyN_reg.fit(X_polyN_train, y_train)

y_train_polyN_pred  = polyN_reg.predict(X_polyN_train)
y_test_polyN_pred   = polyN_reg.predict(X_polyN_train)

r2_polyN = round(r2_score(y_train, y_train_polyN_pred), 4)
label_polyN = "Polynomial "+ str(grado_polinomio) +". R2=" + str(r2_polyN)

# Creation of series for future plot
X_plot_polyN = np.arange(X.min(), X.max(), 1)[:, np.newaxis]
y_pred_polyN = polyN_reg.predict(polyN.fit_transform(X_plot_polyN.reshape(-1,1)))

### plot of polynomial N for entire domain
x_plot = np.arange(0,600,1).reshape(-1,1)
y_plot_polyN = polyN_reg.predict(polyN.fit_transform(x_plot))
plt.figure(figsize = (8, 5))
sns.scatterplot(x=X_train, y=y_train, label="Train")
sns.scatterplot(x=X_test, y=y_test, label="Test")
plt.plot(x_plot, y_plot_polyN, color='cadetblue', linewidth=2, label=label_polyN)
plt.legend()
plt.show()
